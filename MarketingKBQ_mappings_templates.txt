KBQ 1: Most Successful Marketing Channel/Media Partner/Creative Asset
Required Tables:

email_activity - Email campaign performance
marketing_vendor_and_channel - Channel/vendor effectiveness scores
sales_rep_call_activity - Field sales activities
webinar - Webinar attendance

SQL Queries:

1. Email Campaign Performance (by time period):
sqlSELECT 
    Campaign_Name,
    COUNT(*) as emails_sent,
    SUM(CASE WHEN Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) as opens,
    SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) as clicks,
    ROUND(100.0 * SUM(CASE WHEN Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity
WHERE Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
GROUP BY Campaign_Name
ORDER BY click_rate DESC;

2. Marketing Channel/Vendor Rankings:
sqlSELECT 
    Category as channel,
    Vendor,
    ROI,
    Engagement_Rate,
    Total_Reach,
    CPM
FROM marketing_vendor_and_channel
ORDER BY ROI DESC, Engagement_Rate DESC;

3. Sales Channel Activity:
sqlSELECT 
    Call_Activity_Type as channel,
    COUNT(DISTINCT Call_ID) as total_activities,
    COUNT(DISTINCT NPI) as unique_hcps_reached
FROM sales_rep_call_activity
WHERE Call_Date BETWEEN '{start_date}' AND '{end_date}'
GROUP BY Call_Activity_Type
ORDER BY total_activities DESC;

4. Webinar Performance:
sqlSELECT 
    Date_of_Webinar,
    COUNT(DISTINCT npi) as attendees
FROM webinar
WHERE Date_of_Webinar BETWEEN '{start_date}' AND '{end_date}'
GROUP BY Date_of_Webinar
ORDER BY attendees DESC;

KBQ 2: Define Success Metrics
Available Success Metrics by Channel:
Email Channel (email_activity):

Open Rate: Opened_Date_Time IS NOT NULL / total emails
Click Rate (CTR): Clicked_Date_Time IS NOT NULL / total emails
Complaint Rate: Complained_Date_Time IS NOT NULL / total emails
Delivery Rate: Delivered_Date_Time IS NOT NULL / Created emails
Failure Rate: Failed_Date_Time IS NOT NULL / total emails

Marketing Channels (marketing_vendor_and_channel):

ROI Score: 1-7 scale
Engagement Rate: 1-7 scale
Total Reach: 1-7 scale
CPM (Cost Per Thousand): 1-7 scale
Credibility: 1-7 scale
Targeted Audience: 1-7 scale

Sales Activities (sales_rep_call_activity):

Call Volume: Count of calls
HCP Reach: Unique NPIs contacted
Product Detail Rate: Calls with products detailed
Activity Type Distribution: Face-to-face vs meals vs other

Webinar (webinar):

Attendance Count: Total registrants/attendees
HCP Participation Rate: Unique NPIs attending

SQL Query for Multi-Channel Success Metrics:
sql-- Email success metrics
SELECT 
    'Email' as channel,
    COUNT(*) as volume,
    ROUND(100.0 * SUM(CASE WHEN Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as success_rate_1,
    ROUND(100.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as success_rate_2
FROM email_activity
WHERE Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'

UNION ALL

-- Sales activity success metrics
SELECT 
    'Field Sales' as channel,
    COUNT(DISTINCT Call_ID) as volume,
    COUNT(DISTINCT NPI) as success_rate_1,
    COUNT(DISTINCT CASE WHEN Detailed_Products IS NOT NULL THEN Call_ID END) as success_rate_2
FROM sales_rep_call_activity
WHERE Call_Date BETWEEN '{start_date}' AND '{end_date}';

KBQ 3: Campaign Performance Analysis
Required Tables:

email_activity - Campaign metrics and timing
hcp_list - HCP characteristics and priorities
trend_data - Market conditions during campaign
hcp_prescribing_activity - Prescribing behavior correlation

SQL Queries:
1. Campaign Performance by Timing:
sqlSELECT 
    Campaign_Name,
    EXTRACT(DOW FROM Created_Date_Time::timestamp) as day_of_week,
    EXTRACT(HOUR FROM Created_Date_Time::timestamp) as hour_sent,
    COUNT(*) as volume,
    ROUND(100.0 * SUM(CASE WHEN Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity
WHERE Campaign_Name = '{campaign_name}'
GROUP BY Campaign_Name, day_of_week, hour_sent
ORDER BY click_rate DESC;
2. Performance by HCP Characteristics:
sqlSELECT 
    h.Specialty,
    h.CNS_HCP_Priority,
    h.PED_HCP_Priority,
    COUNT(e.*) as emails_sent,
    ROUND(100.0 * SUM(CASE WHEN e.Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(e.*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN e.Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(e.*), 2) as click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.Campaign_Name = '{campaign_name}'
GROUP BY h.Specialty, h.CNS_HCP_Priority, h.PED_HCP_Priority
ORDER BY click_rate DESC;
3. Template/Subject Performance Comparison:
sqlSELECT 
    Template_Name,
    Email_Subject,
    COUNT(*) as usage_count,
    ROUND(AVG(CASE WHEN Opened_Date_Time IS NOT NULL THEN 100.0 ELSE 0 END), 2) as avg_open_rate,
    ROUND(AVG(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 100.0 ELSE 0 END), 2) as avg_click_rate
FROM email_activity
WHERE Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
GROUP BY Template_Name, Email_Subject
HAVING COUNT(*) > 100
ORDER BY avg_click_rate DESC;
4. Market Conditions During Campaign:
sqlSELECT 
    t.year_month,
    t.prod_market,
    SUM(t.trx) as market_trx,
    COUNT(DISTINCT p.prescriber_npi) as active_prescribers
FROM trend_data t
LEFT JOIN hcp_prescribing_activity p 
    ON DATE_TRUNC('month', p.date_submitted::date) = t.year_month::date
WHERE t.year_month BETWEEN '{campaign_start_month}' AND '{campaign_end_month}'
GROUP BY t.year_month, t.prod_market
ORDER BY t.year_month;
5. Campaign vs Historical Benchmark:
sqlWITH campaign_performance AS (
    SELECT 
        Campaign_Name,
        COUNT(*) as volume,
        100.0 * SUM(CASE WHEN Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*) as open_rate,
        100.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*) as click_rate
    FROM email_activity
    GROUP BY Campaign_Name
)
SELECT 
    Campaign_Name,
    open_rate,
    click_rate,
    open_rate - AVG(open_rate) OVER() as open_rate_vs_avg,
    click_rate - AVG(click_rate) OVER() as click_rate_vs_avg
FROM campaign_performance
WHERE Campaign_Name = '{campaign_name}';

KBQ 4: Channel/Tactic ROI Analysis
Required Tables:

marketing_vendor_and_channel - ROI scores by channel
email_activity - Email campaign performance
sales_rep_call_activity - Sales activities
hcp_prescribing_activity - Prescription outcomes
sample_activity - Sample distribution

SQL Queries:
1. Channel ROI Rankings (using scoring system):
sqlSELECT 
    Category as channel,
    Sub_Category as tactic,
    Vendor,
    ROI as roi_score,
    Engagement_Rate,
    Cost_H_M_L as cost_level,
    CPM
FROM marketing_vendor_and_channel
WHERE ROI = (SELECT MAX(ROI) FROM marketing_vendor_and_channel)
ORDER BY ROI DESC, Engagement_Rate DESC;
2. Email Campaign ROI Proxy (Performance/Volume):
sqlSELECT 
    EXTRACT(YEAR FROM Created_Date_Time::date) as year,
    Campaign_Name,
    COUNT(*) as emails_sent,
    SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) as total_clicks,
    ROUND(100.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as ctr,
    -- Proxy ROI: clicks per 1000 emails (assuming similar costs)
    ROUND(1000.0 * SUM(CASE WHEN Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as roi_proxy
FROM email_activity
WHERE EXTRACT(YEAR FROM Created_Date_Time::date) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
GROUP BY year, Campaign_Name
ORDER BY roi_proxy DESC
LIMIT 10;
3. Sales Activity ROI Analysis (HCP Reach Efficiency):
sqlWITH sales_metrics AS (
    SELECT 
        Call_Activity_Type as tactic,
        COUNT(DISTINCT Call_ID) as total_calls,
        COUNT(DISTINCT NPI) as unique_hcps,
        COUNT(DISTINCT CASE WHEN Call_Type = 'Detail with Sample' THEN Call_ID END) as sample_calls
    FROM sales_rep_call_activity
    WHERE EXTRACT(YEAR FROM Call_Date::date) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
    GROUP BY Call_Activity_Type
)
SELECT 
    tactic,
    total_calls,
    unique_hcps,
    ROUND(100.0 * unique_hcps / total_calls, 2) as hcp_efficiency_rate,
    ROUND(100.0 * sample_calls / total_calls, 2) as sample_rate
FROM sales_metrics
ORDER BY hcp_efficiency_rate DESC;
4. Attribution Analysis - Linking Activities to Prescriptions:
sqlWITH activity_prescriptions AS (
    SELECT 
        s.Call_Activity_Type,
        COUNT(DISTINCT s.Call_ID) as activities,
        COUNT(DISTINCT p.prescriber_npi) as prescribing_hcps,
        SUM(p.total_trx) as total_prescriptions
    FROM sales_rep_call_activity s
    JOIN hcp_prescribing_activity p 
        ON s.NPI = p.prescriber_npi 
        AND p.date_submitted BETWEEN s.Call_Date::date AND s.Call_Date::date + INTERVAL '30 days'
    WHERE EXTRACT(YEAR FROM s.Call_Date::date) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
    GROUP BY s.Call_Activity_Type
)
SELECT 
    Call_Activity_Type as tactic,
    activities,
    prescribing_hcps,
    total_prescriptions,
    ROUND(total_prescriptions / activities, 2) as prescriptions_per_activity
FROM activity_prescriptions
ORDER BY prescriptions_per_activity DESC;
5. Sample Distribution ROI:
sqlWITH sample_impact AS (
    SELECT 
        COUNT(DISTINCT s.HCP_Name) as sampled_hcps,
        SUM(s.Quantity) as total_samples,
        COUNT(DISTINCT p.prescriber_npi) as prescribing_sampled_hcps,
        SUM(p.total_trx) as prescriptions_from_sampled
    FROM sample_activity s
    JOIN hcp_list h ON s.HCP_Name = h.First_Name || ' ' || h.Last_Name
    LEFT JOIN hcp_prescribing_activity p 
        ON h.NPI = p.prescriber_npi::integer
        AND p.date_submitted BETWEEN s.Sample_Date::date AND s.Sample_Date::date + INTERVAL '60 days'
    WHERE EXTRACT(YEAR FROM s.Sample_Date::date) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
)
SELECT 
    'Sample Distribution' as channel,
    total_samples,
    sampled_hcps,
    prescribing_sampled_hcps,
    ROUND(100.0 * prescribing_sampled_hcps / sampled_hcps, 2) as conversion_rate,
    ROUND(prescriptions_from_sampled / total_samples, 2) as prescriptions_per_sample
FROM sample_impact;

KBQ 5: Best-Performing Marketing Asset by HCP Type & Patient Type
Required Tables:

email_activity - Email campaign performance
hcp_list - HCP characteristics (specialty, priority)
sales_rep_call_activity - Sales activities by HCP
trend_data - Patient age categories
hcp_prescribing_activity - Prescribing patterns

SQL Queries:
1. Email Performance by HCP Type:
sqlSELECT 
    h.Specialty as hcp_type,
    h.Degree as hcp_degree,
    e.Template_Name as marketing_asset,
    e.Email_Subject,
    COUNT(e.*) as volume,
    ROUND(100.0 * SUM(CASE WHEN e.Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(e.*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN e.Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(e.*), 2) as click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
GROUP BY h.Specialty, h.Degree, e.Template_Name, e.Email_Subject
HAVING COUNT(e.*) > 50
ORDER BY click_rate DESC;
2. Sales Activity Performance by HCP Priority:
sqlSELECT 
    CASE 
        WHEN h.CNS_HCP_Priority IS NOT NULL THEN 'CNS Priority ' || h.CNS_HCP_Priority
        WHEN h.PED_HCP_Priority IS NOT NULL THEN 'PED Priority ' || h.PED_HCP_Priority
        ELSE 'Unclassified'
    END as hcp_segment,
    s.Call_Activity_Type as marketing_asset,
    s.Detailed_Products,
    COUNT(DISTINCT s.Call_ID) as activities,
    COUNT(DISTINCT s.NPI) as unique_hcps
FROM sales_rep_call_activity s
JOIN hcp_list h ON s.NPI = h.NPI
WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
GROUP BY hcp_segment, s.Call_Activity_Type, s.Detailed_Products
ORDER BY activities DESC;
3. Marketing Asset Performance by Patient Type (Adult vs Pediatric):
sqlWITH asset_performance AS (
    SELECT 
        e.Template_Name as asset,
        CASE 
            WHEN e.Template_Name LIKE '%Child%' OR e.Template_Name LIKE '%Ped%' THEN 'Pediatric'
            WHEN h.PED_HCP_Priority IS NOT NULL THEN 'Pediatric'
            ELSE 'Adult'
        END as patient_type,
        COUNT(*) as volume,
        SUM(CASE WHEN e.Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) as clicks
    FROM email_activity e
    LEFT JOIN hcp_list h ON e.HCP_NPI = h.NPI
    WHERE e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY e.Template_Name, patient_type
)
SELECT 
    asset,
    patient_type,
    volume,
    clicks,
    ROUND(100.0 * clicks / volume, 2) as click_rate
FROM asset_performance
WHERE volume > 100
ORDER BY patient_type, click_rate DESC;
4. Product Detail Performance by HCP Specialty:
sqlWITH product_performance AS (
    SELECT 
        h.Specialty,
        CASE 
            WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla XR-ODT'
            WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys XR-ODT'
            WHEN s.Detailed_Products LIKE '%Karbinal%' THEN 'Karbinal ER'
            ELSE 'Other'
        END as product_asset,
        COUNT(DISTINCT s.Call_ID) as details,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(p.total_trx) as resulting_prescriptions
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    LEFT JOIN hcp_prescribing_activity p 
        ON s.NPI = p.prescriber_npi::real
        AND p.product_name LIKE '%' || 
            CASE 
                WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla'
                WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys'
                ELSE ''
            END || '%'
        AND p.date_submitted BETWEEN s.Call_Date::date AND s.Call_Date::date + INTERVAL '30 days'
    WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.Specialty, product_asset
)
SELECT 
    Specialty as hcp_type,
    product_asset,
    details,
    converting_hcps,
    ROUND(100.0 * converting_hcps / NULLIF(details, 0), 2) as conversion_rate,
    ROUND(resulting_prescriptions / NULLIF(details, 0), 2) as prescriptions_per_detail
FROM product_performance
ORDER BY hcp_type, conversion_rate DESC;
5. Combined View - Best Asset by HCP/Patient Segment:
sqlSELECT 
    h.Specialty as hcp_specialty,
    CASE 
        WHEN h.CNS_HCP_Priority = 'A' OR h.PED_HCP_Priority = 'A' THEN 'A-Priority'
        WHEN h.CNS_HCP_Priority = 'B' OR h.PED_HCP_Priority = 'B' THEN 'B-Priority'
        ELSE 'C-Priority'
    END as hcp_priority,
    CASE 
        WHEN h.PED_HCP_Priority IS NOT NULL THEN 'Pediatric'
        ELSE 'Adult'
    END as patient_focus,
    e.Template_Name as top_email_asset,
    MAX(CASE WHEN e.Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) as best_click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
GROUP BY h.Specialty, hcp_priority, patient_focus, e.Template_Name
ORDER BY hcp_specialty, hcp_priority, best_click_rate DESC;

KBQ 6: Channel & Content Resonance by Customer Segment
Required Tables:

hcp_list - Customer segmentation (specialty, priority, location)
email_activity - Email content performance
sales_rep_call_activity - Sales channel engagement
webinar - Webinar participation
sample_activity - Sample acceptance

SQL Queries:
1. Channel Preference by HCP Specialty:
sqlWITH channel_engagement AS (
    -- Email engagement
    SELECT h.Specialty, 'Email' as channel, COUNT(DISTINCT e.HCP_NPI) as engaged_hcps
    FROM email_activity e
    JOIN hcp_list h ON e.HCP_NPI = h.NPI
    WHERE e.Clicked_Date_Time IS NOT NULL 
    AND e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.Specialty
    
    UNION ALL
    
    -- Sales engagement
    SELECT h.Specialty, 'Sales Call' as channel, COUNT(DISTINCT s.NPI) as engaged_hcps
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.Specialty
    
    UNION ALL
    
    -- Webinar engagement
    SELECT h.Specialty, 'Webinar' as channel, COUNT(DISTINCT w.npi) as engaged_hcps
    FROM webinar w
    JOIN hcp_list h ON w.npi = h.NPI
    WHERE w.Date_of_Webinar BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.Specialty
)
SELECT 
    Specialty as customer_segment,
    channel,
    engaged_hcps,
    RANK() OVER (PARTITION BY Specialty ORDER BY engaged_hcps DESC) as channel_rank
FROM channel_engagement
ORDER BY Specialty, channel_rank;
2. Content Performance by HCP Priority Tier:
sqlSELECT 
    CASE 
        WHEN h.CNS_HCP_Priority = 'A' THEN 'CNS A-Priority'
        WHEN h.CNS_HCP_Priority = 'B' THEN 'CNS B-Priority'
        WHEN h.PED_HCP_Priority = 'A' THEN 'PED A-Priority'
        WHEN h.PED_HCP_Priority = 'B' THEN 'PED B-Priority'
        ELSE 'C-Priority'
    END as customer_tier,
    e.Template_Name as content,
    COUNT(*) as exposures,
    ROUND(100.0 * SUM(CASE WHEN e.Opened_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN e.Clicked_Date_Time IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
GROUP BY customer_tier, e.Template_Name
HAVING COUNT(*) > 50
ORDER BY customer_tier, click_rate DESC;
3. Product Interest by Customer Type:
sqlWITH product_engagement AS (
    SELECT 
        h.Specialty,
        h.Degree,
        CASE 
            WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla XR-ODT'
            WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys XR-ODT'
            WHEN s.Detailed_Products LIKE '%Karbinal%' THEN 'Karbinal ER'
            ELSE 'Other'
        END as product_content,
        COUNT(DISTINCT s.Call_ID) as interactions,
        COUNT(DISTINCT s.NPI) as unique_customers
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.Specialty, h.Degree, product_content
)
SELECT 
    Specialty || ' - ' || Degree as customer_profile,
    product_content,
    interactions,
    unique_customers,
    ROUND(interactions::numeric / unique_customers, 2) as avg_interactions_per_customer
FROM product_engagement
WHERE unique_customers > 10
ORDER BY customer_profile, avg_interactions_per_customer DESC;
4. Call Activity Type Preference by Location:
sqlSELECT 
    h.State,
    s.Call_Activity_Type as engagement_type,
    COUNT(DISTINCT s.NPI) as hcps_engaged,
    COUNT(s.Call_ID) as total_activities,
    ROUND(COUNT(s.Call_ID)::numeric / COUNT(DISTINCT s.NPI), 2) as activities_per_hcp
FROM sales_rep_call_activity s
JOIN hcp_list h ON s.NPI = h.NPI
WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
GROUP BY h.State, s.Call_Activity_Type
HAVING COUNT(DISTINCT s.NPI) > 20
ORDER BY h.State, hcps_engaged DESC;
5. Multi-Channel Customer Journey Analysis:
sqlWITH customer_touchpoints AS (
    SELECT 
        h.NPI,
        h.Specialty,
        h.CNS_HCP_Priority,
        h.PED_HCP_Priority,
        COUNT(DISTINCT e.Campaign_ID) as email_campaigns,
        COUNT(DISTINCT s.Call_ID) as sales_calls,
        COUNT(DISTINCT w.Date_of_Webinar) as webinars_attended,
        COUNT(DISTINCT sa.Sample_Date) as sample_interactions
    FROM hcp_list h
    LEFT JOIN email_activity e ON h.NPI = e.HCP_NPI 
        AND e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
    LEFT JOIN sales_rep_call_activity s ON h.NPI = s.NPI 
        AND s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
    LEFT JOIN webinar w ON h.NPI = w.npi 
        AND w.Date_of_Webinar BETWEEN '{start_date}' AND '{end_date}'
    LEFT JOIN sample_activity sa ON h.First_Name || ' ' || h.Last_Name = sa.HCP_Name 
        AND sa.Sample_Date BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY h.NPI, h.Specialty, h.CNS_HCP_Priority, h.PED_HCP_Priority
)
SELECT 
    Specialty as customer_type,
    CASE 
        WHEN email_campaigns > 0 AND sales_calls > 0 THEN 'Multi-Channel'
        WHEN email_campaigns > 0 THEN 'Email-Preferred'
        WHEN sales_calls > 0 THEN 'Sales-Preferred'
        ELSE 'Low-Engagement'
    END as engagement_pattern,
    COUNT(*) as customer_count,
    AVG(email_campaigns) as avg_email_touches,
    AVG(sales_calls) as avg_sales_touches,
    AVG(webinars_attended) as avg_webinars,
    AVG(sample_interactions) as avg_samples
FROM customer_touchpoints
GROUP BY Specialty, engagement_pattern
ORDER BY Specialty, customer_count DESC;

KBQ 7: Channel/Tactic Conversion Analysis
Required Tables:

email_activity - Email engagement tracking
sales_rep_call_activity - Sales touchpoints
hcp_prescribing_activity - Prescription conversions
sample_activity - Sample distribution
webinar - Webinar attendance

SQL Queries:
1. Email Channel Conversions:
sqlWITH email_conversions AS (
    SELECT 
        e.Campaign_Name,
        COUNT(DISTINCT e.HCP_NPI) as clicked_hcps,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(p.total_trx) as total_prescriptions
    FROM email_activity e
    JOIN hcp_prescribing_activity p 
        ON e.HCP_NPI = p.prescriber_npi::real
        AND p.date_submitted BETWEEN e.Clicked_Date_Time::date 
            AND e.Clicked_Date_Time::date + INTERVAL '30 days'
    WHERE e.Clicked_Date_Time IS NOT NULL
        AND e.Created_Date_Time BETWEEN '{start_date}' AND '{end_date}'
    GROUP BY e.Campaign_Name
)
SELECT 
    'Email' as channel,
    Campaign_Name,
    clicked_hcps,
    converting_hcps as conversions,
    ROUND(100.0 * converting_hcps / clicked_hcps, 2) as conversion_rate,
    total_prescriptions
FROM email_conversions
ORDER BY conversions DESC;
2. Sales Call Conversions by Activity Type:
sqlSELECT 
    s.Call_Activity_Type as tactic,
    COUNT(DISTINCT s.NPI) as contacted_hcps,
    COUNT(DISTINCT p.prescriber_npi) as conversions,
    ROUND(100.0 * COUNT(DISTINCT p.prescriber_npi) / COUNT(DISTINCT s.NPI), 2) as conversion_rate,
    SUM(p.total_trx) as total_prescriptions
FROM sales_rep_call_activity s
LEFT JOIN hcp_prescribing_activity p 
    ON s.NPI = p.prescriber_npi::real
    AND p.date_submitted BETWEEN s.Call_Date::date 
        AND s.Call_Date::date + INTERVAL '30 days'
WHERE s.Call_Date BETWEEN '{start_date}' AND '{end_date}'
GROUP BY s.Call_Activity_Type
ORDER BY conversions DESC;
3. Sample Distribution Conversions:
sqlWITH sample_conversions AS (
    SELECT 
        sa.Sample_Product,
        COUNT(DISTINCT sa.HCP_Name) as sampled_hcps,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(sa.Quantity) as samples_distributed,
        SUM(p.total_trx) as prescriptions_generated
    FROM sample_activity sa
    JOIN hcp_list h ON sa.HCP_Name = h.First_Name || ' ' || h.Last_Name
    LEFT JOIN hcp_prescribing_activity p

2. Pre/Post Writing Behavior Analysis:
sqlWITH attendance_cohorts AS (

        npi,

        ON s.NPI = p.prescriber_npi::real
        AND p.date_submitted BETWEEN s.Call_Date::date AND s.Call_Date::date + INTERVAL '30 days'
    WHERE s.Call_Date >= DATE_TRUNC('quarter', CURRENT_DATE - INTERVAL '8 quarters')::date
    GROUP BY quarter
),
productivity_trend AS (
    SELECT 
        quarter,
        total_calls,
        resulting_trx,
        trx_per_call,
        AVG(trx_per_call) OVER (ORDER BY quarter ROWS BETWEEN 3 PRECEDING AND CURRENT ROW) as avg_productivity,
        -- Linear regression slope for trend
        REGR_SLOPE(trx_per_call, EXTRACT(EPOCH FROM quarter)) OVER () as productivity_trend
    FROM rep_productivity
)
SELECT 
    quarter,
    total_calls,
    resulting_trx as actual_sales_volume,
    trx_per_call,
    avg_productivity,
    -- Projected sales = projected calls * projected productivity
    total_calls * (avg_productivity + productivity_trend) as projected_sales_volume
FROM productivity_trend
WHERE quarter >= DATE_TRUNC('quarter', CURRENT_DATE - INTERVAL '4 quarters')::date
ORDER BY quarter;
5. Market Share Projection:
sqlWITH market_trends AS (
    SELECT 
        year_month,
        SUM(CASE WHEN prod_market LIKE '%Cotempla%' OR prod_market LIKE '%Adzenys%' 
            THEN trx ELSE 0 END) as company_trx,
        SUM(CASE WHEN prod_market = 'ADHD MPH Market' THEN trx ELSE 0 END) as total_market_trx,
        ROUND(100.0 * SUM(CASE WHEN prod_market LIKE '%Cotempla%' OR prod_market LIKE '%Adzenys%' 
            THEN trx ELSE 0 END) / NULLIF(SUM(CASE WHEN prod_market = 'ADHD MPH Market' 
            THEN trx ELSE 0 END), 0), 2) as market_share_pct
    FROM trend_data
    WHERE year_month >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '24 months')::date
    GROUP BY year_month
),
share_projection AS (
    SELECT 
        year_month,
        company_trx,
        total_market_trx,
        market_share_pct,
        AVG(market_share_pct) OVER (ORDER BY year_month ROWS BETWEEN 5 PRECEDING AND CURRENT ROW) as avg_share,
        REGR_SLOPE(market_share_pct, EXTRACT(EPOCH FROM year_month::date)) OVER () as share_trend
    FROM market_trends
)
SELECT 
    year_month,
    company_trx as actual_volume,
    market_share_pct as actual_share,
    -- Projected volume = projected market size * projected share
    total_market_trx * (avg_share + share_trend * 3) / 100.0 as projected_volume_3mo,
    avg_share + share_trend * 3 as projected_share_3mo
FROM share_projection
WHERE year_month >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '12 months')::date
ORDER BY year_month;

1. Prescriber Switching Behavior Analysis:
sqlWITH prescriber_product_mix AS (
    SELECT 
        prescriber_npi,
        product_name,
        DATE_TRUNC('month', date_submitted::date) as month,
        SUM(total_trx) as monthly_trx,
        SUM(total_trx) / SUM(SUM(total_trx)) OVER (PARTITION BY prescriber_npi, DATE_TRUNC('month', date_submitted::date)) as product_share
    FROM hcp_prescribing_activity
    WHERE date_submitted >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '12 months')::date
    GROUP BY prescriber_npi, product_name, month
),
switching_indicators AS (
    SELECT 
        prescriber_npi,
        product_name,
        month,
        product_share,
        LAG(product_share) OVER (PARTITION BY prescriber_npi, product_name ORDER BY month) as prev_share,
        product_share - LAG(product_share) OVER (PARTITION BY prescriber_npi, product_name ORDER BY month) as share_change
    FROM prescriber_product_mix
)
SELECT 
    product_name,
    COUNT(DISTINCT CASE WHEN share_change < -0.1 THEN prescriber_npi END) as prescribers_decreasing,
    COUNT(DISTINCT CASE WHEN share_change > 0.1 THEN prescriber_npi END) as prescribers_increasing,
    COUNT(DISTINCT prescriber_npi) as total_prescribers,
    ROUND(AVG(share_change) * 100, 2) as avg_share_change_pct
FROM switching_indicators
WHERE prev_share IS NOT NULL
GROUP BY product_name
ORDER BY prescribers_decreasing DESC;
2. Generic vs Branded Market Dynamics:
sqlWITH generic_brand_trends AS (
    SELECT 
        year_month,
        Product_Gen,
        product_group_name,
        SUM(total_trx) as total_prescriptions,
        LAG(SUM(total_trx), 3) OVER (PARTITION BY Product_Gen ORDER BY year_month) as trx_3mo_ago
    FROM gen1_gen2_liquid_medicaid_trx
    WHERE year_month >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '12 months')::date

)
SELECT 
    year_month,
    Product_Gen as product_generation,
    SUM(total_prescriptions) as current_volume,
    SUM(trx_3mo_ago) as volume_3mo_ago,
    ROUND(100.0 * (SUM(total_prescriptions) - SUM(trx_3mo_ago)) / NULLIF(SUM(trx_3mo_ago), 0), 2) as growth_rate,
    CASE 
        WHEN SUM(total_prescriptions) < SUM(trx_3mo_ago) * 0.9 THEN 'High Switch Risk'
        WHEN SUM(total_prescriptions) < SUM(trx_3mo_ago) * 0.95 THEN 'Medium Switch Risk'
        ELSE 'Low Switch Risk'
    END as switch_risk_indicator
FROM generic_brand_trends
WHERE trx_3mo_ago IS NOT NULL
GROUP BY year_month, Product_Gen
ORDER BY year_month DESC;
3. Market Share Erosion Analysis:
sqlWITH product_market_share AS (
    SELECT 
        year_month,
        prod_market,
        SUM(trx) as product_trx,
        SUM(SUM(trx)) OVER (PARTITION BY year_month) as total_market_trx,
        100.0 * SUM(trx) / SUM(SUM(trx)) OVER (PARTITION BY year_month) as market_share
    FROM trend_data
    WHERE market = 'ADHD'
        AND year_month >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '18 months')::date
    GROUP BY year_month, prod_market
),
share_changes AS (
    SELECT 
        prod_market,
        year_month,
        market_share,
        market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY year_month) as share_change_6mo,
        CASE 
            WHEN market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY year_month) < -2 THEN 'Losing Share'
            WHEN market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY year_month) > 2 THEN 'Gaining Share'
            ELSE 'Stable'
        END as market_position
    FROM product_market_share
)
SELECT 
    prod_market,
    MAX(market_share) as current_share,
    MIN(market_share) as min_share_18mo,
    MAX(market_share) - MIN(market_share) as share_volatility,
    AVG(share_change_6mo) as avg_6mo_change,
    COUNT(CASE WHEN market_position = 'Losing Share' THEN 1 END) as months_losing_share
FROM share_changes
WHERE share_change_6mo IS NOT NULL
GROUP BY prod_market
ORDER BY months_losing_share DESC;
4. Prescriber Loyalty Score:
sqlWITH prescriber_consistency AS (
    SELECT 
        prescriber_npi,
        product_name,
        COUNT(DISTINCT DATE_TRUNC('month', date_submitted::date)) as active_months,
        SUM(total_trx) as total_prescriptions,
        STDDEV(total_trx) as prescription_variability,
        MAX(date_submitted) as last_prescription_date
    FROM hcp_prescribing_activity
    WHERE date_submitted >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '12 months')::date
    GROUP BY prescriber_npi, product_name
),
loyalty_metrics AS (
    SELECT 
        product_name,
        prescriber_npi,
        active_months,
        total_prescriptions,
        CASE 
            WHEN active_months >= 10 AND prescription_variability < 5 THEN 'High Loyalty'
            WHEN active_months >= 6 AND prescription_variability < 10 THEN 'Medium Loyalty'
            ELSE 'Low Loyalty'
        END as loyalty_tier,
        CURRENT_DATE - last_prescription_date::date as days_since_last_rx
    FROM prescriber_consistency
)
SELECT 
    product_name,
    loyalty_tier,
    COUNT(DISTINCT prescriber_npi) as prescriber_count,
    AVG(total_prescriptions) as avg_prescriptions,
    COUNT(CASE WHEN days_since_last_rx > 60 THEN 1 END) as at_risk_prescribers
FROM loyalty_metrics
GROUP BY product_name, loyalty_tier
ORDER BY product_name, loyalty_tier;
5. Payer Coverage Impact on Switching:
sqlWITH payer_coverage_changes AS (
    SELECT 
        product_group_name,
        payer_name,
        plan_name,
        MIN(year_month) as first_coverage_date,
        MAX(year_month) as last_coverage_date,
        COUNT(DISTINCT year_month) as months_covered,
        AVG(total_trx) as avg_monthly_trx,
        STDDEV(total_trx) as trx_volatility
    FROM gen1_gen2_liquid_medicaid_trx
    WHERE year_month >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '12 months')::date
    GROUP BY product_group_name, payer_name, plan_name
)
SELECT 
    product_group_name,
    COUNT(DISTINCT plan_name) as total_plans,
    COUNT(CASE WHEN months_covered < 6 THEN 1 END) as unstable_coverage_plans,
    COUNT(CASE WHEN last_coverage_date < DATE_TRUNC('month', CURRENT_DATE - INTERVAL '3 months')::date THEN 1 END) as recently_dropped_plans,
    AVG(trx_volatility) as avg_volatility,
    CASE 
        WHEN COUNT(CASE WHEN months_covered < 6 THEN 1 END) > 5 THEN 'High Switch Risk - Coverage Issues'
        WHEN AVG(trx_volatility) > 10 THEN 'Medium Switch Risk - Volume Volatility'
        ELSE 'Low Switch Risk'
    END as payer_switch_risk
FROM payer_coverage_changes
GROUP BY product_group_name
ORDER BY unstable_coverage_plans DESC;
