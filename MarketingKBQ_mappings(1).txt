
KBQ 1: 
What was the most successful <marketing channel> or <media partner> or <creative asset> in <time period>?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.

1. Email Campaign Performance: 

SELECT 
    Campaign_Name,
    COUNT(*) as emails_sent,
    SUM(CASE WHEN `Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) as opens,
    SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) as clicks,
    ROUND(100.0 * SUM(CASE WHEN `Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity
WHERE `Created_Date-Time` BETWEEN '2024-07-01' AND '2025-06-30'
GROUP BY Campaign_Name
ORDER BY click_rate DESC;


2. Marketing Channel/Vendor Rankings:

SELECT 
    Category as channel,
    Vendor,
    ROI,
    Engagement_Rate,
    Total_Reach,
    CPM
FROM marketing_vendor_and_channel
ORDER BY ROI DESC, Engagement_Rate DESC;


3. Sales Channel Activity:

SELECT 
    Call_Activity_Type as channel,
    COUNT(DISTINCT Call_ID) as total_activities,
    COUNT(DISTINCT NPI) as unique_hcps_reached
FROM sales_rep_call_activity
WHERE Call_Date BETWEEN '2024-07-01' AND '2025-06-30'
GROUP BY Call_Activity_Type
ORDER BY total_activities DESC;


4. Webinar Performance:

SELECT 
    Date_of_Webinar,
    COUNT(DISTINCT npi) as attendees
FROM webinar
WHERE Date_of_Webinar BETWEEN '2024-07-01' AND '2025-06-30'
GROUP BY Date_of_Webinar
ORDER BY attendees DESC;

KBQ 3: 
Define success - e.g., CTR, open rate, etc?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.


1. Campaign Performance by Timing:

SELECT 
    Campaign_Name,
    DAYOFWEEK(`Created_Date-Time`) as day_of_week,
    HOUR(`Created_Date-Time`) as hour_sent,
    COUNT(*) as volume,
    ROUND(100.0 * SUM(CASE WHEN `Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity
WHERE `Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    AND Campaign_Name IS NOT NULL
GROUP BY Campaign_Name, day_of_week, hour_sent
HAVING COUNT(*) > 50
ORDER BY click_rate DESC
LIMIT 50;


2. Performance by HCP Characteristics:

SELECT 
    h.Specialty, 
    h.CNS_HCP_Priority, 
    h.PED_HCP_Priority, 
    COUNT(*) as emails_sent, 
    ROUND(100.0 * SUM(CASE WHEN e.`Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate, 
    ROUND(100.0 * SUM(CASE WHEN e.`Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate 
FROM email_activity e 
JOIN hcp_list h ON e.HCP_NPI = h.NPI 
WHERE e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    AND e.Campaign_Name IS NOT NULL
GROUP BY h.Specialty, h.CNS_HCP_Priority, h.PED_HCP_Priority 
HAVING COUNT(*) > 10
ORDER BY click_rate DESC;


3. Template/Subject Performance Comparison:

SELECT 
    Template_Name,

FROM email_activity
WHERE `Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'

HAVING COUNT(*) > 100

5. Campaign vs Historical Benchmark:

WITH campaign_performance AS (
    SELECT 
        Campaign_Name,
        COUNT(*) as volume,
        100.0 * SUM(CASE WHEN `Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*) as open_rate,
        100.0 * SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*) as click_rate
    FROM email_activity
    WHERE `Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
        AND Campaign_Name IS NOT NULL
    GROUP BY Campaign_Name
    HAVING COUNT(*) > 100
)
SELECT 
    Campaign_Name,
    volume,
    ROUND(open_rate, 2) as open_rate,
    ROUND(click_rate, 2) as click_rate,
    ROUND(open_rate - AVG(open_rate) OVER(), 2) as open_rate_vs_avg,
    ROUND(click_rate - AVG(click_rate) OVER(), 2) as click_rate_vs_avg
FROM campaign_performance
ORDER BY click_rate_vs_avg DESC
LIMIT 50;

KBQ 4: 
Which <channel> or <tactic> delivered the highest ROI last year?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.


1. Channel ROI Rankings (using scoring system):

SELECT 
    Category as channel,
    Sub_Category as tactic,
    Vendor,
    ROI as roi_score,
    Engagement_Rate,
    Cost_H_M_L as cost_level,
    CPM
FROM marketing_vendor_and_channel
WHERE ROI = (SELECT MAX(ROI) FROM marketing_vendor_and_channel)
ORDER BY ROI DESC, Engagement_Rate DESC;


2. Email Campaign ROI Proxy (Performance/Volume):

SELECT 
    YEAR(`Created_Date-Time`) as year,
    Campaign_Name,
    COUNT(*) as emails_sent,
    SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) as total_clicks,
    ROUND(100.0 * SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as ctr,
    -- Proxy ROI: clicks per 1000 emails (assuming similar costs)
    ROUND(1000.0 * SUM(CASE WHEN `Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as roi_proxy
FROM email_activity
WHERE YEAR(`Created_Date-Time`) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
GROUP BY year, Campaign_Name
ORDER BY roi_proxy DESC
LIMIT 10;


3. Sales Activity ROI Analysis (HCP Reach Efficiency):

WITH sales_metrics AS (
    SELECT 
        Call_Activity_Type as tactic,
        COUNT(DISTINCT Call_ID) as total_calls,
        COUNT(DISTINCT NPI) as unique_hcps,
        COUNT(DISTINCT CASE WHEN Call_Type = 'Detail with Sample' THEN Call_ID END) as sample_calls
    FROM sales_rep_call_activity
    WHERE YEAR(`Call_Date`) = EXTRACT(YEAR FROM CURRENT_DATE) - 1
    GROUP BY Call_Activity_Type
)
SELECT 
    tactic,
    total_calls,
    unique_hcps,
    ROUND(100.0 * unique_hcps / total_calls, 2) as hcp_efficiency_rate,
    ROUND(100.0 * sample_calls / total_calls, 2) as sample_rate
FROM sales_metrics
ORDER BY hcp_efficiency_rate DESC;


4. Attribution Analysis - Linking Activities to Prescriptions:

WITH activity_prescriptions AS (
    SELECT 
        s.Call_Activity_Type,
        COUNT(DISTINCT s.Call_ID) as activities,
        COUNT(DISTINCT p.prescriber_npi) as prescribing_hcps,
        SUM(p.total_trx) as total_prescriptions
    FROM sales_rep_call_activity s
    JOIN hcp_prescribing_activity p 
        ON s.NPI = p.prescriber_npi 
        AND p.date_submitted BETWEEN s.Call_Date AND DATE_ADD(s.Call_Date, INTERVAL 30 DAY)
    WHERE s.Call_Date BETWEEN '2024-01-01' AND '2024-12-31'
    GROUP BY s.Call_Activity_Type
)
SELECT 
    Call_Activity_Type as tactic,
    activities,
    prescribing_hcps,
    total_prescriptions,
    ROUND(total_prescriptions / activities, 2) as prescriptions_per_activity
FROM activity_prescriptions
ORDER BY prescriptions_per_activity DESC;

KBQ 5: 
Identify the best-performing marketing asset by <HCP type> <patient type>

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.


1. Email Performance by HCP Type:

SELECT 
    h.Specialty as hcp_type,
    h.Degree as hcp_degree,
    e.Template_Name as marketing_asset,
    e.Email_Subject,
    COUNT(*) as volume,
    ROUND(100.0 * SUM(CASE WHEN e.`Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN e.`Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
GROUP BY h.Specialty, h.Degree, e.Template_Name, e.Email_Subject
HAVING COUNT(*) > 50
ORDER BY click_rate DESC
LIMIT 100;


2. Sales Activity Performance by HCP Priority:

SELECT 
    CASE 
        WHEN h.CNS_HCP_Priority IS NOT NULL THEN CONCAT('CNS Priority ', h.CNS_HCP_Priority)
        WHEN h.PED_HCP_Priority IS NOT NULL THEN CONCAT('PED Priority ', h.PED_HCP_Priority)
        ELSE 'Unclassified'
    END as hcp_segment,
    s.Call_Activity_Type as marketing_asset,
    s.Detailed_Products,
    COUNT(DISTINCT s.Call_ID) as activities,
    COUNT(DISTINCT s.NPI) as unique_hcps
FROM sales_rep_call_activity s
JOIN hcp_list h ON s.NPI = h.NPI
WHERE s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'
GROUP BY hcp_segment, s.Call_Activity_Type, s.Detailed_Products
ORDER BY activities DESC
LIMIT 100;


3. Marketing Asset Performance by Patient Type (Adult vs Pediatric):

WITH asset_performance AS (
    SELECT 
        e.Template_Name as asset,
        CASE 
            WHEN e.Template_Name LIKE '%Child%' OR e.Template_Name LIKE '%Ped%' THEN 'Pediatric'
            WHEN h.PED_HCP_Priority IS NOT NULL THEN 'Pediatric'
            ELSE 'Adult'
        END as patient_type,
        COUNT(*) as volume,
        SUM(CASE WHEN e.`Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) as clicks
    FROM email_activity e
    LEFT JOIN hcp_list h ON e.HCP_NPI = h.NPI
    WHERE e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY e.Template_Name, patient_type
)
SELECT 
    asset,
    patient_type,
    volume,
    clicks,
    ROUND(100.0 * clicks / volume, 2) as click_rate
FROM asset_performance
WHERE volume > 100
ORDER BY patient_type, click_rate DESC;


4. Product Detail Performance by HCP Specialty:

WITH product_performance AS (
    SELECT 
        h.Specialty,
        CASE 
            WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla XR-ODT'
            WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys XR-ODT'
            WHEN s.Detailed_Products LIKE '%Karbinal%' THEN 'Karbinal ER'
            ELSE 'Other'
        END as product_asset,
        COUNT(DISTINCT s.Call_ID) as details,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(p.total_trx) as resulting_prescriptions
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    LEFT JOIN hcp_prescribing_activity p 
        ON s.NPI = p.prescriber_npi
        AND p.product_name LIKE CONCAT('%', 
            CASE 
                WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla'
                WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys'
                ELSE ''
            END, '%')
        AND p.date_submitted BETWEEN s.Call_Date AND DATE_ADD(s.Call_Date, INTERVAL 30 DAY)
    WHERE s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.Specialty, product_asset
)
SELECT 
    Specialty as hcp_type,
    product_asset,
    details,
    converting_hcps,
    ROUND(100.0 * converting_hcps / NULLIF(details, 0), 2) as conversion_rate,
    ROUND(resulting_prescriptions / NULLIF(details, 0), 2) as prescriptions_per_detail
FROM product_performance
ORDER BY hcp_type, conversion_rate DESC;

KBQ 6: 
What <channels> <content> are most likely to resonate with <X Customer>?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.


1. Channel Preference by HCP Specialty:

WITH channel_engagement AS (
    -- Email engagement
    SELECT h.Specialty, 'Email' as channel, COUNT(DISTINCT e.HCP_NPI) as engaged_hcps
    FROM email_activity e
    JOIN hcp_list h ON e.HCP_NPI = h.NPI
    WHERE e.`Clicked_Date-Time` IS NOT NULL 
    AND e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.Specialty
    
    UNION ALL
    
    -- Sales engagement
    SELECT h.Specialty, 'Sales Call' as channel, COUNT(DISTINCT s.NPI) as engaged_hcps
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    WHERE s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.Specialty
    
    UNION ALL
    
    -- Webinar engagement
    SELECT h.Specialty, 'Webinar' as channel, COUNT(DISTINCT w.npi) as engaged_hcps
    FROM webinar w
    JOIN hcp_list h ON w.npi = h.NPI
    WHERE w.Date_of_Webinar BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.Specialty
)
SELECT 
    Specialty as customer_segment,
    channel,
    engaged_hcps,
    RANK() OVER (PARTITION BY Specialty ORDER BY engaged_hcps DESC) as channel_rank
FROM channel_engagement
WHERE Specialty IS NOT NULL
ORDER BY Specialty, channel_rank;


2. Content Performance by HCP Priority Tier:

SELECT 
    CASE 
        WHEN h.CNS_HCP_Priority = 'A' THEN 'CNS A-Priority'
        WHEN h.CNS_HCP_Priority = 'B' THEN 'CNS B-Priority'
        WHEN h.PED_HCP_Priority = 'A' THEN 'PED A-Priority'
        WHEN h.PED_HCP_Priority = 'B' THEN 'PED B-Priority'
        ELSE 'C-Priority'
    END as customer_tier,
    e.Template_Name as content,
    COUNT(*) as exposures,
    ROUND(100.0 * SUM(CASE WHEN e.`Opened_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as open_rate,
    ROUND(100.0 * SUM(CASE WHEN e.`Clicked_Date-Time` IS NOT NULL THEN 1 ELSE 0 END) / COUNT(*), 2) as click_rate
FROM email_activity e
JOIN hcp_list h ON e.HCP_NPI = h.NPI
WHERE e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
GROUP BY customer_tier, e.Template_Name
HAVING COUNT(*) > 50
ORDER BY customer_tier, click_rate DESC;


3. Product Interest by Customer Type:

WITH product_engagement AS (
    SELECT 
        h.Specialty,
        h.Degree,
        CASE 
            WHEN s.Detailed_Products LIKE '%Cotempla%' THEN 'Cotempla XR-ODT'
            WHEN s.Detailed_Products LIKE '%Adzenys%' THEN 'Adzenys XR-ODT'
            WHEN s.Detailed_Products LIKE '%Karbinal%' THEN 'Karbinal ER'
            ELSE 'Other'
        END as product_content,
        COUNT(DISTINCT s.Call_ID) as interactions,
        COUNT(DISTINCT s.NPI) as unique_customers
    FROM sales_rep_call_activity s
    JOIN hcp_list h ON s.NPI = h.NPI
    WHERE s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'

    interactions,
    unique_customers,
    ROUND(interactions / unique_customers, 2) as avg_interactions_per_customer
FROM product_engagement
WHERE unique_customers > 10

    COUNT(DISTINCT s.NPI) as hcps_engaged,

5. Multi-Channel Customer Journey Analysis:

WITH customer_touchpoints AS (
    SELECT 
        h.NPI,
        h.Specialty,
        h.CNS_HCP_Priority,
        h.PED_HCP_Priority,
        COUNT(DISTINCT e.Campaign_ID) as email_campaigns,
        COUNT(DISTINCT s.Call_ID) as sales_calls,
        COUNT(DISTINCT w.Date_of_Webinar) as webinars_attended,
        COUNT(DISTINCT sa.Sample_Date) as sample_interactions
    FROM hcp_list h
    LEFT JOIN email_activity e ON h.NPI = e.HCP_NPI 
        AND e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    LEFT JOIN sales_rep_call_activity s ON h.NPI = s.NPI 
        AND s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'
    LEFT JOIN webinar w ON h.NPI = w.npi 
        AND w.Date_of_Webinar BETWEEN '2023-01-01' AND '2025-06-30'
    LEFT JOIN sample_activity sa ON CONCAT(h.First_Name, ' ', h.Last_Name) = sa.HCP_Name 
        AND sa.Sample_Date BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.NPI, h.Specialty, h.CNS_HCP_Priority, h.PED_HCP_Priority
)
SELECT 
    Specialty as customer_type,
    CASE 
        WHEN email_campaigns > 0 AND sales_calls > 0 THEN 'Multi-Channel'
        WHEN email_campaigns > 0 THEN 'Email-Preferred'
        WHEN sales_calls > 0 THEN 'Sales-Preferred'
        ELSE 'Low-Engagement'
    END as engagement_pattern,
    COUNT(*) as customer_count,
    ROUND(AVG(email_campaigns), 2) as avg_email_touches,
    ROUND(AVG(sales_calls), 2) as avg_sales_touches,
    ROUND(AVG(webinars_attended), 2) as avg_webinars,
    ROUND(AVG(sample_interactions), 2) as avg_samples
FROM customer_touchpoints
GROUP BY Specialty, engagement_pattern
ORDER BY Specialty, customer_count DESC;

KBQ 7: 
How many conversions came from <channel> or <tactic>?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
All channel performance for the given time period.


1. Email Channel Conversions:

WITH email_conversions AS (
    SELECT 
        e.Campaign_Name,
        COUNT(DISTINCT e.HCP_NPI) as clicked_hcps,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(p.total_trx) as total_prescriptions
    FROM email_activity e
    JOIN hcp_prescribing_activity p 
        ON e.HCP_NPI = p.prescriber_npi
        AND p.date_submitted BETWEEN DATE(e.`Clicked_Date-Time`) 
            AND DATE_ADD(DATE(e.`Clicked_Date-Time`), INTERVAL 30 DAY)
    WHERE e.`Clicked_Date-Time` IS NOT NULL
        AND e.`Created_Date-Time` BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY e.Campaign_Name
)
SELECT 
    'Email' as channel,
    Campaign_Name,
    clicked_hcps,
    converting_hcps as conversions,
    ROUND(100.0 * converting_hcps / clicked_hcps, 2) as conversion_rate,
    total_prescriptions
FROM email_conversions
WHERE clicked_hcps > 10
ORDER BY conversions DESC
LIMIT 50;


2. Sales Call Conversions by Activity Type:

SELECT 
    s.Call_Activity_Type as tactic,
    COUNT(DISTINCT s.NPI) as contacted_hcps,
    COUNT(DISTINCT p.prescriber_npi) as conversions,
    ROUND(100.0 * COUNT(DISTINCT p.prescriber_npi) / COUNT(DISTINCT s.NPI), 2) as conversion_rate,
    SUM(p.total_trx) as total_prescriptions
FROM sales_rep_call_activity s
LEFT JOIN hcp_prescribing_activity p 
    ON s.NPI = p.prescriber_npi
    AND p.date_submitted BETWEEN s.Call_Date 
        AND DATE_ADD(s.Call_Date, INTERVAL 30 DAY)
WHERE s.Call_Date BETWEEN '2023-01-01' AND '2025-06-30'
GROUP BY s.Call_Activity_Type
ORDER BY conversions DESC;


3. Sample Distribution Conversions:

WITH sample_conversions AS (
    SELECT 
        sa.Sample_Product,
        COUNT(DISTINCT sa.HCP_Name) as sampled_hcps,
        COUNT(DISTINCT p.prescriber_npi) as converting_hcps,
        SUM(sa.Quantity) as samples_distributed,
        SUM(p.total_trx) as prescriptions_generated
    FROM sample_activity sa
    JOIN hcp_list h ON sa.HCP_Name = CONCAT(h.First_Name, ' ', h.Last_Name)
    LEFT JOIN hcp_prescribing_activity p 
        ON h.NPI = p.prescriber_npi
        AND p.date_submitted BETWEEN sa.Sample_Date 
            AND DATE_ADD(sa.Sample_Date, INTERVAL 60 DAY)

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
Speaker Program attendance and writing behavior for the given pre/post period


1. Webinar Attendance Frequency Analysis:

WITH hcp_attendance AS (
    SELECT 
        npi,
        COUNT(DISTINCT Date_of_Webinar) as programs_attended,
        MIN(Date_of_Webinar) as first_attendance,
        MAX(Date_of_Webinar) as last_attendance
    FROM webinar
    WHERE Date_of_Webinar BETWEEN '2023-01-01' AND '2025-06-30'
        AND npi IS NOT NULL
    GROUP BY npi
),
prescribing_impact AS (
    SELECT 
        w.npi,
        w.programs_attended,
        SUM(CASE WHEN p.date_submitted < w.first_attendance THEN p.total_trx ELSE 0 END) as pre_attendance_trx,
        SUM(CASE WHEN p.date_submitted >= w.first_attendance THEN p.total_trx ELSE 0 END) as post_attendance_trx
    FROM hcp_attendance w
    LEFT JOIN hcp_prescribing_activity p ON w.npi = p.prescriber_npi
    GROUP BY w.npi, w.programs_attended
)
SELECT 
    programs_attended,
    COUNT(DISTINCT npi) as hcp_count,
    ROUND(AVG(pre_attendance_trx), 2) as avg_pre_trx,

    ROUND(AVG(post_attendance_trx - pre_attendance_trx), 2) as avg_trx_lift,
    ROUND(100.0 * AVG(post_attendance_trx - pre_attendance_trx) / NULLIF(AVG(pre_attendance_trx), 0), 2) as pct_lift
FROM prescribing_impact

        a.npi,
        a.total_programs,
        -- 90 days pre-period
        SUM(CASE WHEN p.date_submitted BETWEEN DATE_SUB(a.first_program_date, INTERVAL 90 DAY) 
            AND a.first_program_date THEN p.total_trx ELSE 0 END) as pre_90d_trx,
        -- 90 days post-period
        SUM(CASE WHEN p.date_submitted BETWEEN a.first_program_date 
            AND DATE_ADD(a.first_program_date, INTERVAL 90 DAY) THEN p.total_trx ELSE 0 END) as post_90d_trx
    FROM attendance_cohorts a
    LEFT JOIN hcp_prescribing_activity p ON a.npi = p.prescriber_npi
    GROUP BY a.npi, a.total_programs
)
SELECT 
    CASE 
        WHEN total_programs = 1 THEN '1 Program'
        WHEN total_programs = 2 THEN '2 Programs'
        WHEN total_programs = 3 THEN '3 Programs'
        WHEN total_programs >= 4 THEN '4+ Programs'
    END as attendance_group,
    COUNT(*) as hcp_count,
    ROUND(AVG(pre_90d_trx), 2) as avg_pre_period_trx,
    ROUND(AVG(post_90d_trx), 2) as avg_post_period_trx,
    ROUND(AVG(post_90d_trx - pre_90d_trx), 2) as avg_lift,
    ROUND(100.0 * AVG(post_90d_trx - pre_90d_trx) / NULLIF(AVG(pre_90d_trx), 0), 2) as pct_change
FROM writing_behavior
WHERE pre_90d_trx > 0  -- Only HCPs who were prescribing before
GROUP BY attendance_group
ORDER BY 
    CASE 
        WHEN attendance_group = '1 Program' THEN 1
        WHEN attendance_group = '2 Programs' THEN 2
        WHEN attendance_group = '3 Programs' THEN 3
        ELSE 4
    END;


3. Optimal Attendance Frequency - Diminishing Returns Analysis:

WITH program_impact AS (
    SELECT 
        w.npi,
        COUNT(DISTINCT w.Date_of_Webinar) as programs_attended,
        SUM(p.total_trx) as total_prescriptions,
        COUNT(DISTINCT p.date_submitted) as prescribing_days
    FROM webinar w
    JOIN hcp_prescribing_activity p 
        ON w.npi = p.prescriber_npi
        AND p.date_submitted BETWEEN w.Date_of_Webinar 
            AND DATE_ADD(w.Date_of_Webinar, INTERVAL 60 DAY)
    GROUP BY w.npi
)
SELECT 
    programs_attended,
    COUNT(*) as hcp_count,
    ROUND(AVG(total_prescriptions), 2) as avg_prescriptions,
    ROUND(AVG(prescribing_days), 2) as avg_active_days,
    ROUND(AVG(total_prescriptions) / programs_attended, 2) as trx_per_program,
    LAG(AVG(total_prescriptions)) OVER (ORDER BY programs_attended) as prev_avg_trx,
    ROUND(AVG(total_prescriptions) - LAG(AVG(total_prescriptions)) OVER (ORDER BY programs_attended), 2) as incremental_impact
FROM program_impact
WHERE programs_attended <= 10
GROUP BY programs_attended
HAVING COUNT(*) >= 5  -- Minimum sample size
ORDER BY programs_attended;


4. Time Between Programs Impact:

WITH program_intervals AS (
    SELECT 
        npi,
        Date_of_Webinar,
        LAG(Date_of_Webinar) OVER (PARTITION BY npi ORDER BY Date_of_Webinar) as prev_program,
        DATEDIFF(Date_of_Webinar, LAG(Date_of_Webinar) OVER (PARTITION BY npi ORDER BY Date_of_Webinar)) as days_between
    FROM webinar
    WHERE npi IS NOT NULL
),
interval_impact AS (
    SELECT 
        i.npi,
        i.days_between,
        SUM(p.total_trx) as prescriptions_after
    FROM program_intervals i
    JOIN hcp_prescribing_activity p 
        ON i.npi = p.prescriber_npi
        AND p.date_submitted BETWEEN i.Date_of_Webinar 
            AND DATE_ADD(i.Date_of_Webinar, INTERVAL 30 DAY)
    WHERE i.days_between IS NOT NULL
    GROUP BY i.npi, i.days_between
)
SELECT 
    CASE 
        WHEN days_between <= 30 THEN '0-30 days'
        WHEN days_between <= 60 THEN '31-60 days'
        WHEN days_between <= 90 THEN '61-90 days'
        WHEN days_between <= 180 THEN '91-180 days'
        ELSE '180+ days'
    END as interval_group,
    COUNT(*) as occurrences,
    ROUND(AVG(prescriptions_after), 2) as avg_post_program_trx
FROM interval_impact
GROUP BY interval_group
ORDER BY 
    CASE 
        WHEN interval_group = '0-30 days' THEN 1
        WHEN interval_group = '31-60 days' THEN 2
        WHEN interval_group = '61-90 days' THEN 3
        WHEN interval_group = '91-180 days' THEN 4
        ELSE 5
    END;

WITH monthly_activity AS (

        SUM(total_trx) as total_rx,
        COUNT(DISTINCT prescriber_npi) as active_prescribers,
        SUM(CASE WHEN total_trx = 1 THEN 1 ELSE 0 END) as new_rx_count
    FROM hcp_prescribing_activity
    WHERE date_submitted BETWEEN DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)
        AND '2025-06-30'
    GROUP BY month

activity_correlation AS (
    SELECT 
        month,
        total_rx,
        new_rx_count,
        active_prescribers,
        LAG(new_rx_count, 1) OVER (ORDER BY month) as prev_month_nrx,

    trailing_3mo_avg,
    active_prescribers,

    -- Budget requirement would be: (target_nbrx / nbrx_per_engagement) * cost_per_engagement
    'Requires cost data for budget calculation' as required_budget

WITH segment_coverage AS (

        h.Specialty,
        h.Degree,
        CASE 
            WHEN h.CNS_HCP_Priority = 'A' THEN 'CNS A-Priority'
            WHEN h.PED_HCP_Priority = 'A' THEN 'PED A-Priority'

    LEFT JOIN webinar w ON h.NPI = w.npi 
        AND w.Date_of_Webinar BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.Specialty, h.Degree, priority_segment

        AND p.date_submitted BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY h.State
)

KBQ 15: 
What are the projected <sales> <expenses? of <period>?

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
Sales forecase vs actuals


1. Historical Sales Trend Analysis (Volume-Based):

WITH monthly_sales AS (
    SELECT 
        DATE_FORMAT(date_submitted, '%Y-%m-01') as month,
        product_name,
        SUM(total_trx) as monthly_trx,
        SUM(total_units) as monthly_units,
        COUNT(DISTINCT prescriber_npi) as active_prescribers
    FROM hcp_prescribing_activity
    WHERE date_submitted >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 24 MONTH)
    GROUP BY month, product_name
),
sales_trends AS (
    SELECT 
        month,
        product_name,
        monthly_trx,
        AVG(monthly_trx) OVER (PARTITION BY product_name ORDER BY month ROWS BETWEEN 11 PRECEDING AND CURRENT ROW) as rolling_12mo_avg,
        monthly_trx - LAG(monthly_trx, 12) OVER (PARTITION BY product_name ORDER BY month) as yoy_change,
        ROUND(100.0 * (monthly_trx - LAG(monthly_trx, 12) OVER (PARTITION BY product_name ORDER BY month)) / 
            NULLIF(LAG(monthly_trx, 12) OVER (PARTITION BY product_name ORDER BY month), 0), 2) as yoy_growth_pct
    FROM monthly_sales
)
SELECT 
    month,
    product_name,
    monthly_trx as actual_volume,
    rolling_12mo_avg,
    yoy_growth_pct,
    -- Simple projection: current volume * (1 + avg growth rate)
    monthly_trx * (1 + AVG(yoy_growth_pct/100.0) OVER (PARTITION BY product_name)) as projected_volume
FROM sales_trends
WHERE month >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)
ORDER BY product_name, month;

3. Activity-Based Expense Drivers:

WITH monthly_activity AS (
    SELECT 
        DATE_FORMAT(Call_Date, '%Y-%m-01') as month,
        COUNT(DISTINCT Call_ID) as sales_calls
    FROM sales_rep_call_activity
    WHERE Call_Date >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)

email_activity_summary AS (

    'Requires unit cost data' as projected_sample_expense

LEFT JOIN email_activity_summary e ON m.month = e.month
LEFT JOIN webinar_summary w ON m.month = w.month
LEFT JOIN sample_summary s ON m.month = s.month

4. Sales Productivity Forecast:

WITH rep_productivity AS (
    SELECT 
        CONCAT(YEAR(s.Call_Date), '-Q', QUARTER(s.Call_Date)) as quarter,
        COUNT(DISTINCT s.Call_ID) as total_calls,
        COUNT(DISTINCT s.NPI) as hcps_reached,
        SUM(p.total_trx) as resulting_trx,

        ON s.NPI = p.prescriber_npi

    WHERE s.Call_Date >= DATE_SUB(CURRENT_DATE, INTERVAL 24 MONTH)
    GROUP BY CONCAT(YEAR(s.Call_Date), '-Q', QUARTER(s.Call_Date))
)
SELECT 
    quarter,
    total_calls,
    resulting_trx as actual_sales_volume,
    trx_per_call,
    AVG(trx_per_call) OVER (ORDER BY quarter ROWS BETWEEN 3 PRECEDING AND CURRENT ROW) as avg_productivity,
    total_calls * AVG(trx_per_call) OVER (ORDER BY quarter ROWS BETWEEN 3 PRECEDING AND CURRENT ROW) as projected_sales_volume

        SUM(CASE WHEN prod_market LIKE '%Cotempla%' OR prod_market LIKE '%Adzenys%' 
            THEN trx ELSE 0 END) as company_trx,
        SUM(CASE WHEN prod_market = 'ADHD MPH Market' THEN trx ELSE 0 END) as total_market_trx,
        ROUND(100.0 * SUM(CASE WHEN prod_market LIKE '%Cotempla%' OR prod_market LIKE '%Adzenys%' 
            THEN trx ELSE 0 END) / NULLIF(SUM(CASE WHEN prod_market = 'ADHD MPH Market' 
            THEN trx ELSE 0 END), 0), 2) as market_share_pct
    FROM trend_data
    WHERE `year_month` >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 24 MONTH)
    GROUP BY `year_month`
),
share_projection AS (
    SELECT 
        `year_month`,
        company_trx,
        total_market_trx,
        market_share_pct,
        AVG(market_share_pct) OVER (ORDER BY `year_month` ROWS BETWEEN 5 PRECEDING AND CURRENT ROW) as avg_share
    FROM market_trends
)
SELECT 
    `year_month`,
    company_trx as actual_volume,
    market_share_pct as actual_share,
    -- Simplified projection without REGR_SLOPE
    total_market_trx * avg_share / 100.0 as projected_volume_3mo,
    avg_share as projected_share_3mo
FROM share_projection
WHERE `year_month` >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)
ORDER BY `year_month`;

KBQ 16: 
Predict existing patient switch propensity to competitive brands (branded or generic).

What inputs are required to generate the analysis needed to create the desired outputs and reports: 
Budget vs actuals


1. Prescriber Switching Behavior Analysis:

WITH prescriber_product_mix AS (
    SELECT 
        prescriber_npi,
        product_name,
        DATE_FORMAT(date_submitted, '%Y-%m-01') as month,
        SUM(total_trx) as monthly_trx,
        SUM(total_trx) / NULLIF(SUM(SUM(total_trx)) OVER (PARTITION BY prescriber_npi, DATE_FORMAT(date_submitted, '%Y-%m-01')), 0) as product_share
    FROM hcp_prescribing_activity
    WHERE date_submitted >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)
    GROUP BY prescriber_npi, product_name, month
),
switching_indicators AS (
    SELECT 
        prescriber_npi,
        product_name,
        month,
        product_share,
        LAG(product_share) OVER (PARTITION BY prescriber_npi, product_name ORDER BY month) as prev_share,
        product_share - LAG(product_share) OVER (PARTITION BY prescriber_npi, product_name ORDER BY month) as share_change
    FROM prescriber_product_mix
)
SELECT 
    product_name,
    COUNT(DISTINCT CASE WHEN share_change < -0.1 THEN prescriber_npi END) as prescribers_decreasing,
    COUNT(DISTINCT CASE WHEN share_change > 0.1 THEN prescriber_npi END) as prescribers_increasing,
    COUNT(DISTINCT prescriber_npi) as total_prescribers,
    ROUND(AVG(share_change) * 100, 2) as avg_share_change_pct
FROM switching_indicators
WHERE prev_share IS NOT NULL
GROUP BY product_name
ORDER BY prescribers_decreasing DESC;


2. Generic vs Branded Market Dynamics:

WITH generic_brand_trends AS (
    SELECT 
        `year_month`,
        Product_Gen,
        product_group_name,
        SUM(total_trx) as total_prescriptions,
        LAG(SUM(total_trx), 3) OVER (PARTITION BY Product_Gen ORDER BY `year_month`) as trx_3mo_ago
    FROM gen1_gen2_liquid_medicaid_trx
    WHERE `year_month` BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY `year_month`, Product_Gen, product_group_name
)
SELECT 
    `year_month`,
    Product_Gen as product_generation,
    SUM(total_prescriptions) as current_volume,
    SUM(trx_3mo_ago) as volume_3mo_ago,
    ROUND(100.0 * (SUM(total_prescriptions) - SUM(trx_3mo_ago)) / NULLIF(SUM(trx_3mo_ago), 0), 2) as growth_rate,
    CASE 
        WHEN SUM(total_prescriptions) < SUM(trx_3mo_ago) * 0.9 THEN 'High Switch Risk'
        WHEN SUM(total_prescriptions) < SUM(trx_3mo_ago) * 0.95 THEN 'Medium Switch Risk'
        ELSE 'Low Switch Risk'
    END as switch_risk_indicator
FROM generic_brand_trends
WHERE trx_3mo_ago IS NOT NULL
GROUP BY `year_month`, Product_Gen
ORDER BY `year_month` DESC;


3. Market Share Erosion Analysis:

WITH product_market_share AS (
    SELECT 
        `year_month`,
        prod_market,
        SUM(trx) as product_trx,
        SUM(SUM(trx)) OVER (PARTITION BY `year_month`) as total_market_trx,
        100.0 * SUM(trx) / SUM(SUM(trx)) OVER (PARTITION BY `year_month`) as market_share
    FROM trend_data
    WHERE market = 'ADHD'
        AND `year_month` >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 18 MONTH)
    GROUP BY `year_month`, prod_market
),
share_changes AS (
    SELECT 
        prod_market,
        `year_month`,
        market_share,
        market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY `year_month`) as share_change_6mo,
        CASE 
            WHEN market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY `year_month`) < -2 THEN 'Losing Share'
            WHEN market_share - LAG(market_share, 6) OVER (PARTITION BY prod_market ORDER BY `year_month`) > 2 THEN 'Gaining Share'
            ELSE 'Stable'
        END as market_position
    FROM product_market_share
)
SELECT 
    prod_market,
    MAX(market_share) as current_share,
    MIN(market_share) as min_share_18mo,
    MAX(market_share) - MIN(market_share) as share_volatility,
    AVG(share_change_6mo) as avg_6mo_change,
    COUNT(CASE WHEN market_position = 'Losing Share' THEN 1 END) as months_losing_share
FROM share_changes
WHERE share_change_6mo IS NOT NULL
GROUP BY prod_market
ORDER BY months_losing_share DESC;


4. Prescriber Loyalty Score:

WITH prescriber_consistency AS (
    SELECT 
        prescriber_npi,
        product_name,
        COUNT(DISTINCT DATE_FORMAT(date_submitted, '%Y-%m-01')) as active_months,
        SUM(total_trx) as total_prescriptions,
        STDDEV(total_trx) as prescription_variability,
        MAX(date_submitted) as last_prescription_date
    FROM hcp_prescribing_activity
    WHERE date_submitted >= DATE_SUB(DATE_FORMAT(CURRENT_DATE, '%Y-%m-01'), INTERVAL 12 MONTH)
    GROUP BY prescriber_npi, product_name
),
loyalty_metrics AS (
    SELECT 
        product_name,
        prescriber_npi,
        active_months,
        total_prescriptions,
        CASE 
            WHEN active_months >= 10 AND prescription_variability < 5 THEN 'High Loyalty'
            WHEN active_months >= 6 AND prescription_variability < 10 THEN 'Medium Loyalty'
            ELSE 'Low Loyalty'
        END as loyalty_tier,
        DATEDIFF(CURRENT_DATE, last_prescription_date) as days_since_last_rx
    FROM prescriber_consistency
)
SELECT 
    product_name,
    loyalty_tier,
    COUNT(DISTINCT prescriber_npi) as prescriber_count,
    ROUND(AVG(total_prescriptions), 2) as avg_prescriptions,
    COUNT(CASE WHEN days_since_last_rx > 60 THEN 1 END) as at_risk_prescribers
FROM loyalty_metrics
GROUP BY product_name, loyalty_tier
ORDER BY product_name, loyalty_tier;


5. Payer Coverage Impact on Switching:

WITH payer_coverage_changes AS (
    SELECT 
        product_group_name,
        payer_name,
        plan_name,
        MIN(`year_month`) as first_coverage_date,
        MAX(`year_month`) as last_coverage_date,
        COUNT(DISTINCT `year_month`) as months_covered,
        AVG(total_trx) as avg_monthly_trx,
        STDDEV(total_trx) as trx_volatility
    FROM gen1_gen2_liquid_medicaid_trx
    WHERE `year_month` BETWEEN '2023-01-01' AND '2025-06-30'
    GROUP BY product_group_name, payer_name, plan_name
)
SELECT 
    product_group_name,
    COUNT(DISTINCT plan_name) as total_plans,
    COUNT(CASE WHEN months_covered < 6 THEN 1 END) as unstable_coverage_plans,
    COUNT(CASE WHEN last_coverage_date < '2024-12-01' THEN 1 END) as recently_dropped_plans,
    ROUND(AVG(trx_volatility), 2) as avg_volatility,
    CASE 
        WHEN COUNT(CASE WHEN months_covered < 6 THEN 1 END) > 5 THEN 'High Switch Risk - Coverage Issues'
        WHEN AVG(trx_volatility) > 10 THEN 'Medium Switch Risk - Volume Volatility'
        ELSE 'Low Switch Risk'
    END as payer_switch_risk
FROM payer_coverage_changes
GROUP BY product_group_name
ORDER BY unstable_coverage_plans DESC;
